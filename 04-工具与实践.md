module.exports = {
  meta: {
    type: 'suggestion',
    docs: { description: '禁止在 useEffect 中使用 async 函数' },
    fixable: null,
  },
  create(context) {
    return {
      CallExpression(node) {
        // 找到 useEffect 调用
        if (isUseEffectCall(node)) {
          const callback = node.arguments[0];
          // 检查回调是否是 async 函数
          if (callback.type === 'ArrowFunctionExpression' &&
              callback.async === true) {
            context.report({
              node: callback,
              message: 'useEffect 回调不应是 async 函数。'
                + '请使用 .then() 或 IIFE 模式。',
            });
          }
        }
      }
    };
  }
};

### 4.1.2 按分析深度分类

```
分析深度层次：

Level 0: 文本/正则匹配
    → grep, git-secrets, detect-secrets

Level 1: AST 模式匹配
    → ESLint, Pylint, Semgrep (基础), Checkstyle

Level 2: 类型感知分析
    → TypeScript, Mypy, Clippy, Error Prone

Level 3: 过程内数据流
    → CodeQL (局部), ESLint type-aware rules, SpotBugs

Level 4: 过程间数据流
    → CodeQL (全局), Infer, Semgrep Pro

Level 5: 深度程序分析
    → 符号执行 (KLEE), 抽象解释 (Astrée), 模型检查 (TLA+)
```

### 4.2.2 GitHub Actions 配置示例

```yaml
name: Quality Gate

on:
  pull_request:
    branches: [main]
  push:
    branches: [main]

jobs:
  # Layer 1: 快速检查（并行）
  fast-checks:
    runs-on: ubuntu-latest
    strategy:
      matrix:
        check: [lint, format, type-check]
    steps:
      - uses: actions/checkout@v4
      - uses: actions/setup-node@v4
        with: { node-version: '20', cache: 'npm' }
      - run: npm ci

      - name: Lint
        if: matrix.check == 'lint'
        run: npx eslint . --max-warnings 0

      - name: Format Check
        if: matrix.check == 'format'
        run: npx prettier --check .

      - name: Type Check
        if: matrix.check == 'type-check'
        run: npx tsc --noEmit

  # Layer 2: 深度分析
  deep-analysis:
    needs: fast-checks
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v4

      - name: Initialize CodeQL
        uses: github/codeql-action/init@v3
        with:
          languages: javascript, python

      - name: Run CodeQL Analysis
        uses: github/codeql-action/analyze@v3

      - name: Run Semgrep
        uses: semgrep/semgrep-action@v1
        with:
          config: >-
            p/security-audit
            p/secrets
          fail-on: error
```

### 4.2.3 渐进式引入策略

```
阶段 1: 观察 (Week 1-2)
    —— 仅报告，不阻断
    —— 建立基线数据
    —— 修复关键 HIGH 问题

阶段 2: 增量阻断 (Week 3-4)
    —— 新增代码不允许引入新 warning
    —— 引入 baseline 机制抑制存量问题
    —— 每周修复一批存量 warning

阶段 3: 全局阻断 (Month 2)
    —— 不允许任何 error 级别问题
    —— warning 数量呈下降趋势
    —— 定期复盘规则有效性

阶段 4: 持续优化 (Ongoing)
    —— 根据误报率调整规则
    —— 引入自定义业务规则
    —— 团队培训，让开发者理解规则背后的原因
```

## 4.3 实战案例

### 4.3.1 案例 1：使用 ESLint 的 Type-Aware 规则检测深坑

**问题**：TypeScript 的 `as` 类型断言可能导致运行时错误。

```typescript
// 危险代码 — as 绕过了类型检查
const data = JSON.parse(response) as User;
console.log(data.email.toUpperCase());
// 如果 data 没有 email 字段，运行时 crash！
```

**ESLint 配置**：
```javascript
// eslint.config.js
import tseslint from 'typescript-eslint';

export default tseslint.config({
  // 启用 type-aware 规则
  languageOptions: {
    parserOptions: {
      project: './tsconfig.json',
      tsconfigRootDir: import.meta.dirname,
    },
  },
  rules: {
    // 必须使用 satisfies 或类型守卫，禁止直接 as
    '@typescript-eslint/consistent-type-assertions': [
      'error',
      { assertionStyle: 'never' }
    ],
    // 禁止 any
    '@typescript-eslint/no-explicit-any': 'error',
    // 强制使用 optional chaining
    '@typescript-eslint/prefer-optional-chain': 'error',
    // 检测未使用的变量（类型感知）
    '@typescript-eslint/no-unused-vars': ['error', {
      argsIgnorePattern: '^_',
      caughtErrors: 'all',
    }],
    // 强制 Promise 被处理
    '@typescript-eslint/no-floating-promises': 'error',
    // 禁止不安全的对象访问
    '@typescript-eslint/no-unsafe-member-access': 'error',
    '@typescript-eslint/no-unsafe-assignment': 'error',
  }
});
```

### 4.3.3 案例 3：自定义 CodeQL 查询检测 SQL 注入

```ql
/**
 * @name 检测 SQL 注入漏洞
 * @description 追踪用户输入是否未经清理地流入 SQL 查询
 * @kind path-problem
 * @id js/sql-injection-custom
 */

import javascript
import DataFlow::PathGraph

class SqlInjectionConfig extends TaintTracking::Configuration {
  SqlInjectionConfig() { this = "SqlInjectionConfig" }

  // 污点源：所有远程输入
  override predicate isSource(DataFlow::Node source) {
    source instanceof RemoteFlowSource
  }

  // 污点汇：SQL 查询执行
  override predicate isSink(DataFlow::Node sink) {
    exists(DataFlow::CallNode call |
      call.getCalleeName() = "query" and
      call.getAnArgument() = sink
    )
  }

  // 清理器：参数化查询绑定、转义函数
  override predicate isSanitizer(DataFlow::Node node) {
    node.getALocalSource().(DataFlow::ParameterNode)
        .getName() = "escaped" or
    node.getALocalSource().getType().toString() = "PreparedStatement"
  }
}

from SqlInjectionConfig cfg, DataFlow::PathNode source,
     DataFlow::PathNode sink
where cfg.hasFlowPath(source, sink)
select sink.getNode(), source, sink,
  "发现 SQL 注入：用户输入 $@ 未经清理流入 SQL 查询。",
  source.getNode(), "从这里来"
```

### 4.3.4 案例 4：ast-grep 大规模重构

```yaml
# ast-grep 规则：将 axios 调用迁移到 fetch
# 文件: migrate-axios-to-fetch.yml

id: axios-to-fetch
language: TypeScript
rule:
  pattern: axios.$METHOD($URL, $CONFIG)
  kind: call_expression

transform:
  FETCH_OPTIONS:
    # 转换配置对象
    replace:
      source: "{ $$$ }"
      replace: "{ $$$ }"
      by: |
        {
          method: '$METHOD',
          $$$CONFIG_BODY
        }

fix: |
  fetch($URL, $FETCH_OPTIONS)
    .then(res => res.json())
```

# 执行重构
ast-grep scan --rule migrate-axios-to-fetch.yml --update-all

### 4.4.2 使用 tree-sitter 构建分析器

```javascript
const Parser = require('tree-sitter');
const JavaScript = require('tree-sitter-javascript');

// 1. 初始化解析器
const parser = new Parser();
parser.setLanguage(JavaScript);

// 2. 解析代码
const sourceCode = `
function calculateTotal(items) {
  let total = 0;
  for (const item of items) {
    total = total + item.price * item.quantity;
    debugger; // BAD!
  }
  return total;
}
`;

const tree = parser.parse(sourceCode);

// 3. 使用 S-表达式查询
const query = new Parser.Query(
  JavaScript,
  `(debugger_statement) @debugger`
);

const matches = query.matches(tree.rootNode);
matches.forEach((match) => {
  const node = match.captures[0].node;
  console.log(
    `[WARN] 发现 debugger 语句: ` +
    `行 ${node.startPosition.row + 1} ` +
    `列 ${node.startPosition.column + 1}`
  );
});
// 输出: [WARN] 发现 debugger 语句: 行 6 列 4
```

### 4.4.3 使用 TypeScript Compiler API

```typescript
import ts from 'typescript';

function analyzeFile(fileName: string) {
  const program = ts.createProgram([fileName], {
    target: ts.ScriptTarget.ES2022,
    strict: true,
  });

  const sourceFile = program.getSourceFile(fileName)!;
  const checker = program.getTypeChecker();

  // 遍历 AST
  function visit(node: ts.Node) {
    // 检测: 使用了 any 类型
    if (ts.isVariableDeclaration(node) && node.type === undefined) {
      const symbol = checker.getSymbolAtLocation(node.name);
      if (symbol) {
        const type = checker.getTypeOfSymbolAtLocation(symbol, node);
        if (checker.typeToString(type) === 'any') {
          const { line, character } = sourceFile.getLineAndCharacterOfPosition(
            node.name.getStart()
          );
          console.log(
            `[WARN] ${fileName}:${line + 1}:${character + 1} ` +
            `— 变量 "${node.name.getText()}" 类型为 any`
          );
        }
      }
    }

    // 检测: 使用了非空断言 (!)
    if (ts.isNonNullExpression(node)) {
      const { line, character } = sourceFile.getLineAndCharacterOfPosition(
        node.getStart()
      );
      console.log(
        `[INFO] ${fileName}:${line + 1}:${character + 1} ` +
        `— 使用了非空断言 (!)，可能掩盖 null 问题`
      );
    }

    ts.forEachChild(node, visit);
  }

  visit(sourceFile);
}
```

## 4.5 最佳实践清单

### 4.5.1 规则设计原则

| 原则 | 说明 |
|------|------|
| **低误报率** | 误报 > 10% 时开发者会忽略所有告警 |
| **可操作** | 每条规则必须能指导开发者如何修复 |
| **有文档** | 解释"为什么"比"是什么"更重要 |
| **可自动修复** | 优先选择能 auto-fix 的规则 |
| **上下文感知** | 避免跨文件无上下文的规则 |
| **可配置** | 允许项目级别定制阈值和排除模式 |

### 4.5.2 团队采用策略

```
1. 从零规则开始
   └── 而不是从"全部启用"开始

2. 一次引入一组相关规则
   └── 例如：先用 @typescript-eslint/recommended

3. 监控数据
   ├── 误报率
   ├── 规则触发频率
   └── 修复率

4. 定期修剪
   ├── 关闭误报率 > 20% 的规则
   └── 增加新的上下文相关规则

5. 开发者培训
   └── 每周分享一个"被静态分析捕获的 bug"案例
```

### 4.5.3 性能优化

| 策略 | 效果 |
|------|------|
| **缓存** | ESLint cache、TS incremental builds |
| **增量分析** | 仅分析变更文件（lint-staged） |
| **并行化** | 多 worker 并行执行 |
| **选择更快的工具** | Ruff > Flake8, Oxlint > ESLint |
| **分级运行** | 快速检查在 PR，慢速检查在 nightly |
| **结果缓存** | CodeQL 结果缓存、Semgrep findings 持久化 |

## 本章小结

- 工具选择需根据**语言、分析深度、性能、团队规模**综合决策
- **渐进式引入 + 增量阻断**是最有效的推行策略
- 规则设计核心原则：**低误报、可操作、有文档**
- 性能优化关键：**缓存、增量、并行、选择正确工具**
- **tree-sitter** 和 **ast-grep** 正在成为通用代码分析基础设施
- AI 增强分析是下一波浪潮，但确定性的规则分析仍是基石
